\begin{problem}{6}
  Let $\Sigma$ = \{0,1\}
  \begin{enumerate}
    \item Let A = $\{ 0^ku0^k | k \ge 1$ and $u \in \Sigma^* \}$ Show that A is regular \\
    \item Let B = $\{ 0^k1u0^k | k \ge 1$ and $u \in \Sigma^* \}$. Show that B is not regular.
  \end{enumerate}
\end{problem}
